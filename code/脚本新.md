# 脚本新

```bash
# 进入镜像所在目录
cd ./report-back

# 查找符合命名规则的 tar 文件（仅取第一个）
tar_file=$(ls report-manage-test_*.tar 2>/dev/null | head -n1)
if [ -z "$tar_file" ]; then
    echo "未找到 report-manage-test_*.tar 文件"
    exit 1
fi
echo "找到镜像文件: $tar_file"

# 从文件名提取 compileId（去掉前缀 report-manage-test_ 和后缀 .tar）
compileId=$(echo "$tar_file" | sed -n 's/^report-manage-test_\(.*\)\.tar$/\1/p')
if [ -z "$compileId" ]; then
    echo "无法从文件名提取 compileId，请检查文件名格式"
    exit 1
fi
echo "提取的 compileId: $compileId"

# 加载镜像，并捕获原始镜像名称
load_output=$(docker load -i "$tar_file")
original_image=$(echo "$load_output" | sed -n 's/^Loaded image: \(.*\)$/\1/p')
if [ -z "$original_image" ]; then
    echo "无法获取加载的镜像名称，请检查 docker load 输出"
    exit 1
fi
echo "原始镜像: $original_image"

# 构造目标镜像完整名称
target_image="cr-ee.registry.cn-nanjing-jszygs-d01.res.cloud.jszygs.com/gs-zngk-acr/report-manage:${compileId}"

# 重新打标签
docker tag "$original_image" "$target_image"

# 登录镜像仓库
docker login --username xzyc-zngk-test-1@1023264075126523 --password-stdin < ../docker_passwd.txt cr-ee.registry.cn-nanjing-jszygs-d01.res.cloud.jszygs.com

# 推送镜像
docker push "$target_image"

# 记录推送后的镜像名供回调使用
imageName="$target_image"

# 执行回调（注意使用双引号确保变量展开）
curl -X POST \
    -F "imageName=${imageName}" \
    -F "crid=${crid}" \
    -F "compileId=${compileId}" \
    -F "appName=${appName}" \
    https://yunxiao.jszygs.com/package-switch/callBack

# 清理本地镜像（删除新标签和原始镜像）
docker rmi "$target_image"
docker rmi "$original_image"
```
